function timeToString(time) {
let second = (time / 1000) % 60; //秒
let minu = ((time / 1000 - second) / 60) % 60;
let hours = ((time / 1000 - second) / 60 - minu) / 60;

return `${hours}时 ${minu}分 ${second}秒`;
}

function computerTime(list) {
let workTime = 0;
let weekTime = 0;
let allTime = 0;

const detailList = [];

for (let day of list) {
const { work_day, datetypename, sb_dk_time, xb_dk_time } = day;
let time8 = new Date(`${work_day} 08:00:00`);
let time9 = new Date(`${work_day} 09:00:00`);

if (sb_dk_time && xb_dk_time) {
let minStartTime = new Date(`${work_day} 17:30:00`);
let maxStartTime = new Date(`${work_day} 18:30:00`);
let sbdkTime = new Date(sb_dk_time);
let xbdkTime = new Date(xb_dk_time);

if (datetypename === "工作日") {
//时间比8点早 开始时间为17:30
//时间比9点晚 开始时间为18:30
//在8-9之间 17:30+比八点多的时间
let startTime;
if (new Date(sb_dk_time) <= time8) { startTime=minStartTime; } else if (new Date(sb_dk_time)>= time9) {
  startTime = maxStartTime;
  } else {
  startTime = new Date(minStartTime.getTime() + (sbdkTime - time8));
  }

  detailList.push({
  ...day,
  type: 0,
  typename: "平时",
  startTime: startTime,
  sum: xbdkTime - startTime,
  sumString:
  xbdkTime - startTime > 0 ? timeToString(xbdkTime - startTime) : 0,
  });
  } else if (datetypename === "休息日" || datetypename === "法定节日") {
  //直接结束时间减去开始时间
  detailList.push({
  ...day,
  type: 1,
  typename: "周末",
  startTime: sbdkTime,
  sum: xbdkTime - sbdkTime,
  sumString:
  xbdkTime - sbdkTime > 0 ? timeToString(xbdkTime - sbdkTime) : 0,
  });
  } else {
  //调休、年假等等
  //如果有开始打开时间且小于18:00，开始时间为18:30
  if (sbdkTime < new Date(`${work_day} 18:00:00`)) { detailList.push({ ...day, type: 0, typename: "平时" , startTime:
    maxStartTime, sum: xbdkTime - maxStartTime, sumString: xbdkTime - maxStartTime> 0
    ? timeToString(xbdkTime - maxStartTime)
    : 0,
    });
    }
    }
    }
    }

    detailList.forEach((item) => {
    if (item.type === 0 && item.sum > 0) {
    workTime = workTime + item.sum;
    }

    if (item.type === 1 && item.sum > 0) {
    weekTime = weekTime + item.sum;
    }
    });

    allTime = workTime + weekTime;

    console.table({
    工作日: timeToString(workTime),
    周末: timeToString(weekTime),
    全部: timeToString(allTime),
    });

    console.table(
    detailList.map((item) => {
    return {
    日期: item.work_day,
    开始时间: item.sb_dk_time,
    结束时间: item.xb_dk_time,
    加班开始: item.startTime,
    类型: item.typename,
    加班时长: item.sumString,
    };
    })
    );
    }
